Pseudocode — audit-micro-skill-scripts.py

def main():
    stage_files = glob("skills/workspace-hub/stages/stage-NN-*.md")
    items = []
    for f in sorted(stage_files):
        stage_num = extract_stage_number(f)
        lines = read_checklist_lines(f)  # lines starting with digit. or - [ ]
        for line in lines:
            cls = classify(line)  # scriptable / judgment / already-scripted
            items.append({stage, line, cls, priority_score})

    report = {
        summary: {total, scriptable_count, judgment_count, already_scripted_count},
        high_priority: [items where score >= threshold],
        all_items: items
    }
    write_yaml(report, "assets/WRK-1144/micro-skill-script-candidates.yaml")

def classify(line):
    if calls_existing_script(line):  return "already-scripted"
    if has_binary_keywords(line):    return "scriptable"   # verify, check, exists, count, run
    return "judgment"

def priority_score(item):
    score = 0
    if item.stage in [1, 5, 7, 17]:  score += 3   # hard-gate stages
    if appears_in_N_or_more_stages(item, 3): score += 2
    if item.cls == "scriptable":     score += 1
    return score
