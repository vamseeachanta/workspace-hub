Mission

Implement a Cathodic Protection (CP) design and verification module for offshore fixed platforms per DNV-RP-B401-2021. Cover current demand calculation, anode sizing, resistance calculation, and adequacy verification across three structural zones (submerged, splash, atmospheric) and three anode types (flush-mounted, stand-off, bracelet). Plug the gap flagged in WRK-383.

Out of scope: DNV-RP-F103 pipeline CP (handled separately), FEniCSx integration (deferred), ICCP systems.

_b401_surface_areas(inputs) → {zone_id: area_m2, total_m2: float}
    # Parse inputs.structure.zones → zone area map; validate uniqueness, positivity
    for each zone in inputs.structure.zones:
        assert zone.zone unique, zone.area_m2 >= 0, zone.zone field present
        areas[zone.zone] = zone.area_m2
    areas["total_m2"] = sum(areas.values())
    raise ValueError if total == 0 or any zone is duplicate / missing

_b401_coating_breakdown(inputs, design_life) → {zone_id: {f_ci, f_cm, f_cf}}
    # Compute Cat I/II/III/bare coating breakdown at initial/mean/final time
    # B401-2021 Sec 3.4.6: f_c(t) = f_ci + k * t   (clamped at 1.0)
    T = design_life  # years
    for each zone:
        params = COATING_CATEGORIES[zone.coating_category]  # {f_ci, k}
        f_cm = min(params.f_ci + params.k * T/2, 1.0)   # mean
        f_cf = min(params.f_ci + params.k * T,   1.0)   # final
    raise ValueError if unknown category, T <= 0, or NaN

_b401_current_densities(inputs) → {zone_id: {i_mean_A_m2, base_zone, temperature_band}}
    # Look up mean design current density from Table 3-1 by zone and seawater temp
    # Submerged: 4 temp bands (≤7 / >7-12 / >12-17 / >17 °C)
    # Splash/atmospheric: temperature-independent
    temp_band = classify(inputs.environment.seawater_temperature_C)
    for each zone:
        base_zone = zone.base_zone ?? zone.zone  # supports segmented zones
        coated_or_bare = "bare" if cat == "bare" else "coated"
        i_mean = TABLE_3_1[base_zone][temp_band][coated_or_bare]
    raise ValueError if unknown zone, unknown category, or NaN temp

_b401_current_demand(inputs, areas, densities, breakdown) → {zone_id: {...}, total_mean_A, total_final_A}
    # I = A * i_mean * f_c per zone; accumulate totals (full precision)
    for each zone:
        I_mean  = areas[zone] * densities[zone].i_mean_A_m2 * breakdown[zone].f_cm
        I_final = areas[zone] * densities[zone].i_mean_A_m2 * breakdown[zone].f_cf
    return per-zone detail + total_mean_A + total_final_A

_b401_anode_resistance(inputs) → float  # ohms
    # Anode-to-electrolyte resistance via Dwight formula (B401-2021 Sec 4.9)
    if type in (flush_mounted, stand_off):
        R = (rho / 2πL) * (ln(4L/r) - 1)   # Dwight; require 4L/r > e
    elif type == bracelet:
        R = (rho / 2πL) * (ln(2πL/r) - 1)  # Modified Dwight; require 2πL/r > e
    raise ValueError if unknown type, rho/L/r <= 0, or geometry too stubby

_b401_anode_requirements(inputs, current_demand) → {total_mass_kg, anode_count, ...}
    # Anode mass and count sizing per B401-2021 Sec 4.3
    # M = (I_mean * T * 8760) / (epsilon * u)   [kg]
    # N = ceil(M / m_anode)
    epsilon = ANODE_CAPACITY[material]  # 2000 Ah/kg (Al), 780 Ah/kg (Zn)
    M = current_demand.total_mean_A * T * 8760 / (epsilon * u)
    N = ceil(M / individual_anode_mass_kg)
    raise ValueError if material unknown, T <= 0, u not in (0,1], m_a <= 0

_b401_verify_current_output(inputs, anode_req, resistance, current_demand) → {adequate: bool, ...}
    # Check if installed anodes provide enough current at end-of-life (governing case)
    E_drive = E_structure(-0.80V) - E_anode(-1.05V Al | -1.00V Zn)   # driving voltage
    I_per_anode = E_drive / R_a
    I_total = N * I_per_anode
    adequate = I_total >= current_demand.total_final_A
    recommended_count = max(N_mass, ceil(I_final / I_per_anode))  # satisfies both criteria
