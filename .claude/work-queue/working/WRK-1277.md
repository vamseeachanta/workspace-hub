Path in index is `/mnt/remote/ace-linux-2/dde/documents/...` — needs remapping.

```bash
# On ace-linux-2:
cd ~/workspace-hub

# Reclassify error PDFs with path remapping
uv run --no-project python -c "
import json, os, sys, time, signal

index_path = 'data/document-index/index.jsonl'
tmp_path = 'data/document-index/index.jsonl.tmp'

# Path remap: NFS mount path → local path on ace-linux-2
PATH_REMAP = {
    '/mnt/remote/ace-linux-2/dde/': '/home/vamsee/dde/',
    '/mnt/remote/ace-linux-2/': '/home/vamsee/',
}

def remap_path(p):
    for remote, local in PATH_REMAP.items():
        if p.startswith(remote):
            return p.replace(remote, local, 1)
    return p

import pdfplumber

class TimeoutError(Exception):
    pass

def timeout_handler(signum, frame):
    raise TimeoutError('pdfplumber timeout')

def classify(path, max_pages=3, timeout_sec=15):
    local_path = remap_path(path)
    if not os.path.exists(local_path):
        return 'error', 'file_not_found_local'
    if os.path.getsize(local_path) == 0:
        return 'empty', 'zero_bytes'
    signal.signal(signal.SIGALRM, timeout_handler)
    signal.alarm(timeout_sec)
    try:
        with pdfplumber.open(local_path) as pdf:
            n = len(pdf.pages)
            if n == 0:
                return 'empty', 'no_pages'
            indices = list(range(min(n, max_pages)))
            with_text = sum(
                1 for i in indices
                if len((pdf.pages[i].extract_text() or '').strip()) >= 50
            )
            ratio = with_text / len(indices)
            if ratio >= 0.8: return 'machine', ''
            elif ratio >= 0.2: return 'mixed', ''
            else: return 'ocr-needed', ''
    except TimeoutError:
        return 'error', 'pdfplumber_timeout'
    except Exception as e:
        msg = str(e).lower()
        if 'fopn' in msg or 'encrypt' in msg or 'password' in msg:
            return 'drm-protected', type(e).__name__
        return 'error', f'{type(e).__name__}: {str(e)[:60]}'
    finally:
        signal.alarm(0)

t0 = time.time()
fixed = 0
total = 0
from collections import Counter
new_rd = Counter()

with open(index_path) as fin, open(tmp_path, 'w') as fout:
    for line in fin:
        rec = json.loads(line)
        total += 1
        if rec.get('readability') == 'error' and rec.get('ext') == 'pdf':
            path = rec.get('path', '')
            rd, reason = classify(path)
            rec['readability'] = rd
            if reason:
                rec['readability_error'] = reason
            else:
                rec.pop('readability_error', None)
            fixed += 1
            new_rd[rd] += 1
        fout.write(json.dumps(rec, separators=(',', ':')) + '\n')
        if total % 50000 == 0:
            elapsed = time.time() - t0
            print(f'  [{total:,}] {fixed:,} fixed ({elapsed:.0f}s)', flush=True)

os.rename(tmp_path, index_path)
elapsed = time.time() - t0
print(f'Done in {elapsed:.0f}s ({elapsed/3600:.1f}h)')
print(f'Reclassified: {fixed:,}')
for k, v in new_rd.most_common():
    print(f'  {k}: {v:,}')
"
```
