Code: auto-sync-queue.sh

# Monitoring with inotifywait
inotifywait -m -r -e modify,move,create,delete --format '%w%f' "$QUEUE_DIR" | while read -r FILE
do
    # Ignore hidden files or temporary files
    if [[ "$(basename "$FILE")" =~ ^\. ]]; then continue; fi
    
    # 5-second debounce loop logic
    run_sync() {
        local now=$(date +%s)
        if (( now - LAST_SYNC < 5 )); then return; fi
        "$SYNC_SCRIPT" auto >> "$LOG_FILE" 2>&1
        LAST_SYNC=$(date +%s)
    }
    run_sync
done
